                            }
                        }
                    }
                }
                \TR*{$\lor$}
                \pstree{\TR*{$K$}\nput{20}{\pssucc}{\color{blue}$p = \str{and f t}$}}{
                    \pstree{\TR*{$K$}\nput{160}{\pssucc}{\color{blue}$p = \str{f}$}}{
